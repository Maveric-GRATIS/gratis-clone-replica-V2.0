// VideoPlayer Component
function VideoPlayer() {
  const [isPlaying, setIsPlaying] = useState(false);
  const containerRef = useRef<HTMLDivElement>(null);
  const isInView = useInView(containerRef, { once: true, margin: '-100px' });
  
  return (
    <motion.div
      ref={containerRef}
      initial={{ opacity: 0, y: 40 }}
      animate={isInView ? { opacity: 1, y: 0 } : {}}
      transition={{ duration: 0.6 }}
      className="relative aspect-video max-w-4xl mx-auto rounded-2xl overflow-hidden shadow-2xl"
    >
      {!isPlaying ? (
        <div className="relative w-full h-full">
          <Image
            src="/images/video-poster.jpg"
            alt="GRATIS explainer video"
            fill
            className="object-cover"
          />
          <div className="absolute inset-0 bg-jet-black/30 flex items-center justify-center">
            <motion.button
              whileHover={{ scale: 1.1 }}
              whileTap={{ scale: 0.95 }}
              onClick={() => setIsPlaying(true)}
              className="w-24 h-24 rounded-full bg-hot-lime flex items-center justify-center shadow-glow-lime"
            >
              <Play className="w-10 h-10 text-jet-black ml-1" fill="currentColor" />
            </motion.button>
          </div>
        </div>
      ) : (
        <MuxPlayer
          playbackId={process.env.NEXT_PUBLIC_MUX_EXPLAINER_PLAYBACK_ID}
          autoPlay
          className="w-full h-full"
          accentColor="#C1FF00"
        />
      )}
    </motion.div>
  );
}

## MUX CONFIGURATION (src/lib/mux/client.ts)

import Mux from '@mux/mux-node';

export const mux = new Mux(
  process.env.MUX_TOKEN_ID,
  process.env.MUX_TOKEN_SECRET
);

export const { Video, Data } = mux;

export async function createVideoAsset(input: string, options?: {
  passthrough?: string;
  playbackPolicy?: 'public' | 'signed';
}) {
  const asset = await Video.Assets.create({
    input,
    playback_policy: [options?.playbackPolicy || 'public'],
    passthrough: options?.passthrough,
    mp4_support: 'standard',
    master_access: 'temporary',
  });
  return asset;
}

## Key Functions:

1. getOrCreateStripeCustomer(userId, email, name) - Creates/retrieves Stripe customer
2. createMembershipCheckout(options) - Creates checkout for TRIBE membership
3. createDonationCheckout(options) - Creates checkout for donations with allocation
4. createEventTicketCheckout(options) - Creates checkout for event tickets
5. cancelSubscription(subscriptionId) - Cancels subscription at period end
6. createCustomerPortalSession(customerId, returnUrl) - Opens customer portal

## Webhook Handler (src/app/api/webhooks/stripe/route.ts)

Handles these events:
- checkout.session.completed → handleMembershipCheckout, handleDonationCheckout, handleEventTicketCheckout
- customer.subscription.updated → updates user membership status
- customer.subscription.deleted → downgrades user to explorer
- invoice.payment_failed → marks subscription as past_due
